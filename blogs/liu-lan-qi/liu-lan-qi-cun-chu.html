    # Storage

    • WebStorage主要提供了一种机制，可以让浏览器提供一种比cookie更直观的key,value存储方式;
      • localStorage:本地存储，提供的是一种永久性的存储方法，在关闭掉网页重新打开时，存储的内容依然保留;
      • sessionStorage: 会话存储，提供的是本次会话的存储，在关闭掉会话时，存储的内容会被清除; (每打开一个标签页打开，关闭标签页清除)
      • Storage的存储的value是字符串
    localStorage.setItem("name", obj);
const obj = localStorage.getItem("name")

    # 封装Storage工具类

    /*
	可存储session或loacl
*/

class MYCache {
	constructor(isLocal = true){
        this.storage = isLocal ? localStorage : sessionStorage;
    }	
    
    setItem(key, value){}	
		if(value){
            this.storage.setItem(key, JSON.stringify(value));
        }
    }
    getItem(key){
        let value = this.storage.getItem(key);
        if(value){
            return JSON.parse(value);
        }
    }
    removeItem(key){
    	this.storage.removeItem (key)
    }
    clear() {
    	this.storage.clear()
    }
    key (index)(
    	return this.storage.key(index)
	}
    length() {
    	return this.storage.length
    }

const localCache = new MYCache()
const sessionCache = new MYCache(false)

export {
    localCache,
    sessionCache
}

    # IndexedDB

    如果有大量数据需要存储在客户端，可以使用IndexDB

    • IndexDB是一种底层的API，用于在客户端存储大量的结构化数据。
      • 它是一种事务型数据库系统，是一种基于JavaScript面向对象数据库，有点类似于NoSQL （非关系型数据库）;
      • IndexDB本身就是基于事务的，我们只需要指定数据库模式，打开与数据库的连接，然后检索和更新一系列事务即可;

    # 使用

    // 打开数据库，有自动打开，没有就创建

const 
dbRequest= indexedDB.open("sql");

dbRequest.onerror = function(err){
    //打开失败回调
}
let db = null;
dbRequest.onsuccess = function(event){
    //打开成功回调
    db = event.target.result;	//
}

//第一次打开/版本更新
dbRequest.onupgradeneeded = function(event){
    //可以在这个回调创建存储对象
    const db = event.target.result;
    db.createObjectStore("user", {keyPath: "id"});	//可以理解为建表
}
