    装饰器模式

    月夕花辰

    # 装饰器模式

    通过给一个实例化对象添加方法而不是给整个类添加方法，更加灵活也可以让类不那么复杂

    # 例子

    let duck = {
	makeVoice: function() {
        console.log('发出声音')
    }
    sleep: function() {
        console.log('睡觉')
    }
	walk: function() {
        console.log('走')
    }
	init: {
        this.makeVoice();
        this.sleep();
        this.walk();
    }
}

after(fn, afterFn) {
    return function() {
        const _ = fn.apply(this, arguments);
        after.apply(this, arguments);
        return _;
    }
}

addFunction() {
    console.log('下蛋');
}

//通过after函数，返回一个函数，外部apply修改this指向目标对象，返回函数内部通过apply执行对象内部函数和新加函数
after(duck.init, addFunction).apply(duck);

    # 单例模式

    function Window(name) {
    this.name = name;
}
Window.prototype.getName = function () {
    console.log(this.name);
}


let CreateWindow = (function () {
    let instance;
    return function (name) {
        if (!instance) {
            instance = new Window(name);
        }
        return instance;
    }
})();
let w1 = new CreateWindow('zfpx1');
let w2 = new CreateWindow('zfpx2');
console.log(w1 === w2);
