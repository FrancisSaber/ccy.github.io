//定义Promise的三个常量
const PROMISE_STATUS_PENDING = 'pending' 
const PROMISE_STATUS_FULFILLED = 'fulfilled'
const PROMISE_STATUS_REJECTED = 'rejected'
class HYPromise{
    constructor(executor){
        this.status = PROMISE_STATUS_PENDING;
        this.value = void 0;
        this.reason = void 0;
        this.onFuilfilledFns = [];
        this.onRejectedFns = [];
        //resolve和reject负责改变promise状态和触发回调，回调的触发放到微队列执行
        const resolve = (value) =>{			//resovle要主动触发then内的响应回调
            if(this.status === PROMISE_STATUS_PENDING){
               queueMicrotask(() => {		//将该函数加入微任务队列
                    if(this.status !== PROMISE_STATUS_PENDING) return //解决res和rej同时执行的问题
                    this.status = PROMISE_STATUS_FULFILLED;
                    this.value = value;			//获取回调
                    this.onFuilfilledFns.forEach(e=>{
                        e(this.value);
                    })
               })
           }
        }
        const reject = (reason) => {
            if(this.status === PROMISE_STATUS_PENDING){  
                queueMicrotask(() => {
                    if(this.status !== PROMISE_STATUS_PENDING) return
                    this.status = PROMISE_STATUS_REJECTED;
                    this.reason = reason;		//获取回调
                    this.onRejectedFns.forEach(e=>{
                        e(this.reason);
                    })
                })
            }   
        }
        //executor是我们传入的函数
        try{			//为了处理直接抛出错误的情况
            executor(resolve, reject);
        }catch(err){
            reject(err);
        }
    }
    //then方法负责将成功回调和失败回调挂载到promise对象上，且实现then方法的多次调用
    then(onfulfilled, onRejected){		//定义then函数
        return new HYPromise((resolve, reject) => {
            //1.如果then调用状态已经确定，可以直接调用,当.then在宏任务队列执行的时候
            if(this.status === PROMISE_STATUS_FULFILLED&&onfulfilled){
            		onfulfilled(thia.value);
        	}
        	if(this.status === PROMISE_STATUS_REJECTED&&onfulfilled){
            	onRejected(thia.reason);
        	}
            //一下是为了实现then的链式调用，.then的返回值需要是一个promise对象，这个返回的promise对象结果如何要根据一下逻辑判断决定，且一列代码同时也执行了其所接收的promise的对应回调
            if(this.status === PROMISE_STATUS_PENDING){
                //通过try catch 语法，不管执行成功回调还是失败回调都执行resolve，以回调函数的返回值为参数，只有捕获到错误才执行reject，以错误原因为参数
                //为了可以获取回调函数的返回值，往数组加入的函数改造为在箭头函数内执行以获取返回值，之所以把回掉放入一个数组是为了实现对同一个promise的重复then调用
            	this.onFuilfilledFns.push(()=>{
                    try{
                        const value = onfulfilled(this.value);
                        resolve(value)
                    }catch(err{
                        reject(err);
                    }
                });
                this.onRejectedFns.push(() => {
                    try{
                        const reason = onRejected(this.reason);
                        resolve(reason);
                    }catch(err){
                        reject(err);
                    }
                });
        	 }
        })		
    }
}

//工具函数，捕获真正的异常
functon exexFunctionWithCatchError(execFn, value, resolve, reject){
    try{
        const result = execFn(value);	//执行传入的回调函数
        resolve(value);	//不管成功还是失败回调都执行resolve函数，为了实现then的返回值是一个promise
    }catch(err){
        reject(err);
    }
}
class MPromise{
    constructor(executor){
        this.status = 'pending'
        this.value = void 0;
        this.reason = void 0;		
        this.onFuilfilledFns = [];	//存储成功回调
        this.onRejectedFns = [];	//存储失败回调
        const resolve = (value) => {
            if(this.status === 'pending'){
                queueMicrotask(()=>{
                    if(this.status !== 'pending')return
                    this.value = value;
                    this.status = 'fulfilled'
                    this.onFuilfilledFns.forEach(e=>e(this.value));
                })
            }
        }
        const reject = (reason) =>{
            if(this.status === 'pending'){
                queueMicrotake(() => {
                    if(this.status !== 'pending')return
                    this.reason = reason;
                    this.status = 'reject'
                    this.onRejectedFns.forEach(e=>e(this.reason));
                })
            }
        }
        executor(resolve, reject);		// executor = 外界传入(resolve, reject) => {}
    }
    then(onFulfilled, onRejected){
        //如果没有传失败回调，失败回调为直接抛出一个错误
        onRejected = onRejected || (err => {throw err})     
        //如果没有传成功回调，成功回调返回resolve结果
        onFulfilled = onFulFilled || (value => {return value}) 
        //执行then会返回new一个promise对象并返回，且这个promise的executor是会立即z
        return new MPromise(resolve, reject){
            if(this.status === 'fulfilled'){
                exexFunctionWithCatchError(onFulfilled, this.value, resolve, reject)
            }
            if(this.status === 'reject'){
                exexFunctionWithCatchError(onRejected, this.reason, resolve, reject)
            }
            if(this.status === 'pending'){
                this.onFuilfilledFns.push(()=>{
                    execFunctionWithCatchError(onFulfilled, this.value, resolve, reject)
                })
                this.onRejectedFns.push(()=>{
                    execFunctionWithCatchError(onRejected, this.reason, resolve, reject)
                })  
            }
        }
    }
    catch(onRejected){
        return this.then(undefined, onRejected);
    }
    finally(onFinally){
        //直接调用then方法，不管成功还是失败都调用同个回调
        this.then(()=>{
            onFinally()
        }, () => {
            onFinally()
        })
    }
}

class MYPromise{
    static resoleve(value){
        return new MYPromise((resolve) => {
            resolve(value)
        })
    }
    static reject(reason){

    static all(list) {			//所有promise都成功resolve一个数组，只要有一个失败直接执行对应reject
        return new MYPromise((resolve, rejected) => {
            let count = 0;
            const value = [];
                list.forEach(promise => {
                    promise.then(res=>{
                        value.push(res);
                        count++;
                        if(count === list.length){
                            resolve(value)
                        }
                    }, err=>{
                        reject(err)
                    })
                })
            }
        })
    }
    static allSettled(promises){					//resolve或reject一个promise结果数组,
        return new MYPromise((resolve, reject)=>{
            const results = [];
            promises.forEach(promise=>{
                promise.then(res=>{
                    results.push({status: 'fulfilled', value: res})
                	if(results.length === promises.length){
                    	resolve(results);
                	}
                },err=>{
                     results.push({status: 'rejected', value: err})
            		if(results.length === promises.length){
                    	resolve(results)
            		}
                })
            }
        })
    }
	static race(promise){					//执行最先完成的promise，无论成功或失败
         return new Promise((resolve, reject)=>{
             promise.forEach(promise =>{
                 promise.then(res=>{
                     resolve(res)
                 },err=>{
                     resolve(err)
                 })
             })
         })   
    }
    static any(promises){
        //1.必须等到至少一个成功才返回
        //2.如果全部失败执行失败回调，以所有promise的失败信息数组为参数
        return new Promise((resolve, reject)=>{
            const reasons = []
            promises.forEach(promise=>{
                promise.then(resolve,err=>{
                	reasons.push(err)
                	if(reasons.length === promises.length){
                    	reject(reasons)
                	}
            	})
            })
        })
    }

    # 五.Promise静态类方法

    # all

    • 必须等待所有promise都有结果才执行
    • 只要有一个promise会失败就执行失败回调

    # allSettled

    • 所有都要结果，一定执行resolve
    • 其把成功和失败的

    # race

    • 只要有结果就执行响应回调，无论成功还是失败

    # any

    • 必须等到一个成功回调
    • 如果全部失败才执行失败回调
